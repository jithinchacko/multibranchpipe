pipeline {
    agent { label 'master'}
    
    parameters {
        string (name: 'BUILDMETHOD', defaultValue: 'clean package', description: 'build method in maven')
    }
    stages {
        stage('scm') {
            steps {
                git 'https://github.com/jithinchacko/mavenjavacode.git'
            }
        }
        stage('build') {
            steps{
                sh "mvn ${params.BUILDMETHOD}"
            }
        }
        stage('postbuild') {
            steps{
                archiveArtifacts "target/*.jar"
                stash name: 'jarfile', includes: 'target/*.jar'
            }
        }
        stage('copy to some other location'){
            steps{
                unstash name: 'jarfile'
            }
        }
    }
}
